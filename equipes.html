import { db } from './firebase-init.js'; import { collection, onSnapshot, query } from "https://www.gstatic.com/firebasejs/10.12.5/firebase-firestore.js"; const roundsBody = document.getElementById('roundsBody'); const rankingEl = document.getElementById('ranking'); const lastResult = document.getElementById('lastResult'); const teamCount = document.getElementById('teamCount'); function normalizeRound(round) { const r = String(round || '').trim().toUpperCase(); if (r === 'TESTE' || r === 'ROUND TESTE') return 'teste'; if (r === '1' || r === 'ROUND 1' || r === 'ROUND OFICIAL 1') return 'oficial1'; return r; } function bestVisibleTableLabel(result) { return result?.tableName ? ` • ${result.tableName}` : ''; } function buildRanking(results) { const byTeam = {}; results.forEach((r) => { if (!r.teamId) return; if (!byTeam[r.teamId]) { byTeam[r.teamId] = { name: r.teamName || 'Equipe sem nome', teste: null, oficial1: null, publicScore: 0, publicTable: '', updated: r.createdAtLocal || '' }; } const roundKey = normalizeRound(r.round); if (roundKey !== 'teste' && roundKey !== 'oficial1') return; const current = byTeam[r.teamId][roundKey]; if (!current || Number(r.score || 0) > Number(current.score || 0)) { byTeam[r.teamId][roundKey] = r; } if (new Date(r.createdAtLocal || 0) > new Date(byTeam[r.teamId].updated || 0)) { byTeam[r.teamId].updated = r.createdAtLocal; } }); Object.values(byTeam).forEach((team) => { const officialScore = Number(team.oficial1?.score || 0); team.publicScore = officialScore; team.publicTable = team.oficial1?.tableName || ''; }); return Object.values(byTeam).sort((a, b) => b.publicScore - a.publicScore || a.name.localeCompare(b.name)); } function renderRoundsTable(ranking) { if (!roundsBody) return; roundsBody.innerHTML = ranking.length ? ranking.map((team) => { const testeTable = bestVisibleTableLabel(team.teste); const officialTable = bestVisibleTableLabel(team.oficial1); return ` ${team.name} ${team.teste?.score ?? '-'}${testeTable ? `${testeTable.replace(' • ', '')}` : ''} ${team.oficial1?.score ?? '-'}${officialTable ? `${officialTable.replace(' • ', '')}` : ''} ${team.publicScore} pts `; }).join('') : 'Aguardando resultados...'; } function renderRanking(ranking) { if (teamCount) teamCount.textContent = `${ranking.length} equipes`; if (!rankingEl) return; rankingEl.innerHTML = ''; } function renderLastVisibleResult() { if (lastResult) lastResult.innerHTML = ''; } onSnapshot(query(collection(db, 'results')), (snap) => { const results = []; snap.forEach((d) => results.push({ id: d.id, ...d.data() })); results.sort((a, b) => new Date(b.createdAtLocal || 0) - new Date(a.createdAtLocal || 0)); const ranking = buildRanking(results); const count = ranking.length; document.body.classList.toggle('dashboard-many', count > 6); document.body.classList.toggle('dashboard-lots', count > 10); renderRanking(ranking); renderRoundsTable(ranking); renderLastVisibleResult(); });
